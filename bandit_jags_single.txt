# Model for a restless bandit task:
# Kalman filter with Probability of Maximum Utility choice rule
# as per Konstantinidis & Speekenbrink (2015)
model{
  # Choice data
  for (j in 1:nBlocks){
    for (k in 1:nTrials){
      
      d[j,k] ~ dmnorm(mu[j,k,1:nBandits], sigma[j,k,1:nBandits])
      
    }
  }
}
